    private void AnalyzeImageWithGemini(string filePath, out bool isFake, out string detectedDept)
    {
        isFake = false;
        detectedDept = "Sanitation"; // Default

        try
        {
            ServicePointManager.SecurityProtocol = (SecurityProtocolType)3072;

            byte[] imageBytes = File.ReadAllBytes(filePath);
            string base64Image = Convert.ToBase64String(imageBytes);

            string url = "https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=" + geminiApiKey;

            // 🔥 Highly strict prompt preventing false positives
            string prompt = "You are a strict city municipal image classifier. Analyze this image carefully. " +
                            "Rule 1: If the image primarily shows a human face, a selfie, a car/motorcycle/vehicle, an animal (dog/cat/cow), a screen (monitor/phone), inside of a room/house, or a blank wall, YOU MUST CLASSIFY IT AS 'FAKE'. " +
                            "Rule 2: If the image shows a genuine civic issue (broken road, garbage dump, broken street light, broken pipe, open manhole), classify it as 'REAL'. " +
                            "Rule 3: If REAL, identify the department. Broken poles, wires, streetlights = 'ELECTRIC'. Leaking water, open manholes, broken pipes = 'WATER'. Potholes, garbage, dead animals, dirty roads = 'SANITATION'. " +
                            "Output Format EXACTLY like this (NO markdown, NO extra words): FAKE:TRUE/FALSE DEPT:ELECTRIC/WATER/SANITATION";

            string jsonPayload = "{" +
                "\"contents\": [{" +
                    "\"parts\": [" +
                        "{\"text\": \"" + prompt + "\"}," +
                        "{\"inline_data\": {\"mime_type\": \"image/jpeg\",\"data\": \"" + base64Image + "\"}}" +
                    "]" +
                "}]" +
            "}";

            HttpWebRequest request = (HttpWebRequest)WebRequest.Create(url);
            request.Method = "POST";
            request.ContentType = "application/json";

            using (StreamWriter writer = new StreamWriter(request.GetRequestStream()))
            {
                writer.Write(jsonPayload);
            }

            using (HttpWebResponse response = (HttpWebResponse)request.GetResponse())
            using (StreamReader reader = new StreamReader(response.GetResponseStream()))
            {
                string rawResult = reader.ReadToEnd().ToUpper();

                // Clean the output heavily
                string cleanResult = Regex.Replace(rawResult, @"[^A-Z:]", "");

                // Check Fake Status
                if (cleanResult.Contains("FAKETRUE") || cleanResult.Contains("FAKE:TRUE"))
                {
                    isFake = true;
                }
                else
                {
                    // Strict Department Routing
                    if (cleanResult.Contains("DEPTELECTRIC") || cleanResult.Contains("DEPT:ELECTRIC"))
                    {
                        detectedDept = "Electric";
                    }
                    else if (cleanResult.Contains("DEPTWATER") || cleanResult.Contains("DEPT:WATER"))
                    {
                        detectedDept = "Water";
                    }
                    else
                    {
                        // Safe fallback for genuine issues that don't match Water/Electric
                        detectedDept = "Sanitation";
                    }
                }
            }
        }
        catch
        {
            isFake = false;
            detectedDept = "Sanitation";
        }
    }
